~~~typescript
// 아래 예제 활용을 알아보기 전에 never의 특징을 알아보자
// !!! 유니언('|')은 A or B를 나타내고 교차('&')는 모두를 만족하는 새로운 타입을 나타낸다. !!!

// 0 + number = number 유니언('|')에서 never는 없어짐
type resUnion = never | string; // string
// 0 * number = 0 교차('&')에서는 덮어쓴다
type resCross = never & string; // never
~~~

~~~typescript
type aPerson = {
  name: 'aName';
  a: number;
}

type bPerson = {
  name: 'bName';
  b: number;
}
~~~

## 2. 읽기 전용 속성
~~~typescript
type TReadOnly = {
  readonly body : {
    readonly height : number;
    weight: number;
  }

  name: string;
  age: number;
}
let person1: TReadOnly = {
  name: 'gayoung',
  age: 26,
  body: {
    height: 160,
    weight: 0
  }
}

person1.age = 27; // 통과
person1.name = 'kong'; // 오류 
// person1을 const로 선언해버리면 되지 않나요? -> 네 됩니다.
// 변수영역에서 사용하는 const, let, 객체 속성영역에 사용하는 readonly

person1.body.weight = 100; // 통과
person1.body.height = 203; // 오류 // readonly도 객체처럼 1depth 까지만 적용된다
~~~

## 3. 선택적 속성 
~~~typescript
type TFunc = (a: number, b?: number) => void;

const func1: TFunc = (a) => {
    console.log(a);
}

const func2: TFunc = (a, b) => {
    if (b !== undefined) {
        console.log(a, b);
    } else {
        console.log(a);
    }
};
~~~
